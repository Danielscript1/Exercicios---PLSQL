11)-----declare o cursor------
declare
cursor cursor_emp is
select first_name || ' ' || last_name name, salary
from employees;
----fim da declaração-----

----declaração da variavel para trabalhar com cursor
var_simples cursor_emp%rowtype;
begin 
 
 ---se o curso não estiver aberto abra o cursor ----
 if not  cursor_emp%isopen THEN
 open cursor_emp;
 end if;
 
 
 ---execute o loop ate não encontra mais cursor-----
 loop
 FETCH cursor_emp into var_simples;
 ---saia do cursor ate não estiver mais registro para processar
 exit WHEN cursor_emp%notfound;
 DBMS_OUTPUT.PUT_LINE('nome: '||var_simples.name || ' salario: '||var_simples.salary );
 end loop;
 close cursor_emp;
